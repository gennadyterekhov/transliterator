About API


right now there's only one endpoint available  
POST /api/transliterate  
Content-Type: application/json

possible "from" values: ["cyrillic"]
possible "to" values: ["latin", "arabic"]
        
        request example:  
        
            {
                "from": "cyrillic",
                "to": "latin",
                "text": "привет"
            }
        
        response example:
        
            {
                "ok": true,
                "from": "cyrillic",
                "to": "latin",
                "text": "privet"
            }
        
            request example:  
            
                {
                    "from": "cyrillic",
                    "to": "arabic",
                    "text": "привет"
                }
            
            response example:
            
                {
                    "ok": true,
                    "from": "cyrillic",
                    "to": "arabic",
                    "text": "پريڤعت"
                }
